## 变更概述

```javascript
/**
 * Toolbar Plugin
 * Adds toolbar UI to diagrams
 */

export const ToolbarPlugin = {
    name: 'toolbar',
    version: '1.0.0',

    afterRender({ diagram, code, renderer }) {
        const wrapper = diagram.parentElement;
        if (!wrapper) return;

        // wrapper 需要 position:relative 以支持工具栏绝对定位
        if (getComputedStyle(wrapper).position === 'static') {
            wrapper.style.position = 'relative';
        }

        const toolbar = document.createElement('div');
        toolbar.className = 'mermaid-toolbar';
        toolbar.style.cssText = `
            position: absolute; top: 8px; right: 8px; z-index: 10;
            display: flex; gap: 4px;
            opacity: 0; transition: opacity 0.2s ease;
            pointer-events: none;
        `;

        wrapper.addEventListener('mouseenter', () => {
            toolbar.style.opacity = '1';
            toolbar.style.pointerEvents = 'auto';
        });
        wrapper.addEventListener('mouseleave', () => {
            toolbar.style.opacity = '0';
            toolbar.style.pointerEvents = 'none';
        });

        wrapper.appendChild(toolbar);
    }
};
```

**说明**：实现 `afterRender` 钩子，在每个 `diagram.parentElement`（`.mermaid-diagram-wrapper`）内创建绝对定位工具栏容器，通过 `mouseenter`/`mouseleave` 事件控制工具栏显示/隐藏，CSS `transition` 实现平滑过渡。

```javascript
/**
 * Fullscreen Plugin
 * Adds fullscreen viewing capability
 */

function createToolbarButton(icon, ariaLabel) {
    const btn = document.createElement('button');
    btn.className = 'mermaid-toolbar-btn';
    btn.setAttribute('aria-label', ariaLabel);
    btn.style.cssText = `
        background: var(--bg-secondary, #1e293b);
        border: 1px solid var(--border, rgba(255,255,255,0.1));
        border-radius: 4px; padding: 4px 6px;
        cursor: pointer; font-size: 14px; color: var(--text-primary, #e2e8f0);
        line-height: 1;
    `;
    btn.textContent = icon;
    return btn;
}

function openFullscreen(diagram) {
    const svg = diagram.querySelector('svg');
    if (!svg) return;

    const overlay = document.createElement('div');
    overlay.className = 'mermaid-fullscreen-overlay';
    overlay.style.cssText = `
        position: fixed; inset: 0; z-index: 9999;
        background: rgba(0,0,0,0.85);
        display: flex; align-items: center; justify-content: center;
    `;

    const clonedSvg = svg.cloneNode(true);
    clonedSvg.style.cssText = 'max-width:90vw; max-height:90vh; width:auto; height:auto;';
    overlay.appendChild(clonedSvg);

    const closeBtn = createToolbarButton('✕', '关闭全屏');
    closeBtn.style.position = 'absolute';
    closeBtn.style.top = '16px';
    closeBtn.style.right = '16px';
    overlay.appendChild(closeBtn);

    const close = () => {
        document.removeEventListener('keydown', escHandler);
        document.body.removeChild(overlay);
    };
    const escHandler = (e) => { if (e.key === 'Escape') close(); };

    closeBtn.addEventListener('click', close);
    overlay.addEventListener('click', (e) => { if (e.target === overlay) close(); });
    document.addEventListener('keydown', escHandler);
    document.body.appendChild(overlay);
}

export const FullscreenPlugin = {
    name: 'fullscreen',
    version: '1.0.0',

    afterRender({ diagram, code, renderer }) {
        const wrapper = diagram.parentElement;
        if (!wrapper) return;

        const toolbar = wrapper.querySelector('.mermaid-toolbar');
        if (!toolbar) return;

        const btn = createToolbarButton('⛶', '全屏查看');
        btn.addEventListener('click', () => openFullscreen(diagram));
        toolbar.appendChild(btn);
    }
};
```

**说明**：实现全屏模态逻辑，使用 `position:fixed; inset:0` 创建全屏遮罩，SVG 通过 `cloneNode(true)` 克隆（保留原图），支持 ESC/关闭按钮/遮罩点击三种退出方式，`removeEventListener` 清理 ESC 监听避免内存泄漏。

**变更后**（来源：设计文档#关键代码说明）：

```javascript
/**
 * Download Plugin
 * Adds download as PNG capability
 */

function createToolbarButton(icon, ariaLabel) {
    const btn = document.createElement('button');
    btn.className = 'mermaid-toolbar-btn';
    btn.setAttribute('aria-label', ariaLabel);
    btn.style.cssText = `
        background: var(--bg-secondary, #1e293b);
        border: 1px solid var(--border, rgba(255,255,255,0.1));
        border-radius: 4px; padding: 4px 6px;
        cursor: pointer; font-size: 14px; color: var(--text-primary, #e2e8f0);
        line-height: 1;
    `;
    btn.textContent = icon;
    return btn;
}

function downloadPNG(diagram) {
    const svg = diagram.querySelector('svg');
    if (!svg) return;

    const svgStr = new XMLSerializer().serializeToString(svg);
    const svgBlob = new Blob([svgStr], { type: 'image/svg+xml;charset=utf-8' });
    const url = URL.createObjectURL(svgBlob);

    const img = new Image();
    img.onload = () => {
        const canvas = document.createElement('canvas');
        const dpr = window.devicePixelRatio || 1;
        canvas.width = (img.width || 800) * dpr;
        canvas.height = (img.height || 600) * dpr;

        const ctx = canvas.getContext('2d');
        ctx.fillStyle = '#0f172a';
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        ctx.scale(dpr, dpr);
        ctx.drawImage(img, 0, 0);
        URL.revokeObjectURL(url);

        canvas.toBlob((blob) => {
            if (!blob) return;
            const a = document.createElement('a');
            a.href = URL.createObjectURL(blob);
            a.download = `mermaid-diagram-${Date.now()}.png`;
            a.click();
            URL.revokeObjectURL(a.href);
        }, 'image/png');
    };
    img.src = url;
}

export const DownloadPlugin = {
    name: 'download',
    version: '1.0.0',

    afterRender({ diagram, code, renderer }) {
        const wrapper = diagram.parentElement;
        if (!wrapper) return;

        const toolbar = wrapper.querySelector('.mermaid-toolbar');
        if (!toolbar) return;

        const btn = createToolbarButton('⬇', '下载 PNG');
        btn.addEventListener('click', () => downloadPNG(diagram));
        toolbar.appendChild(btn);
    }
};
```

**说明**：实现 SVG→Canvas→PNG 导出链路，通过 `XMLSerializer` 序列化 SVG，`Image` + `Canvas` 完成位图转换，`canvas.toBlob` 异步导出 PNG（相比 `toDataURL` 内存占用更低），`devicePixelRatio` 支持 HiDPI 高清导出，所有 Blob URL 使用后立即释放。
